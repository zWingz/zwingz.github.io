then

vim ~/.bash_profile or ~/.zshrc or ~/.profile or ~/.bashrc

insert

export NVM_DIR="$HOME/.nvm" [ -s "$NVM_DIR/nvm.sh" ] && . "$NVM_DIR/nvm.sh" # This loads nvm

  # 本地node版本
  nvm ls

  nvm ls-remote
  # 安装最新node(稳定版,即偶数版)
  nvm install --lts
  # 安装指定版本node
  nvm install v8.9.4
  # 切换到指定版本/别名node
  nvm use v8.9.4
  # 给node版本起别名
  nvm alias default v8.9.4
